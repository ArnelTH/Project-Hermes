struct HermesHeader {
  let magic: UInt32      // Un nombre magique pour identifier que c'est bien notre protocole (ex: 0x48524D53 pour "HRMS")
  let version: UInt16    // La version du protocole (ex: 1) pour assurer la compatibilité future
  let type: UInt16       // Le type de message (ex: Commande, Réponse, Événement, etc.)
  let length: UInt32     // La taille du reste du message (le payload)
  let timestamp: UInt64  // L'horodatage pour éviter les attaques par rejeu (replay attacks)
  let nonce: UInt64      // Un nombre aléatoire unique pour la cryptographie
}

// En Swift, une enum peut être adossée à un type binaire spécifique (ici UInt16 pour prendre 2 octets)
enum MessageType: UInt16 {
    case command = 1
    case response = 2
    case event = 3
}

struct Message {
    let header: HermesHeader
    let payload: Data        // "Data" en Swift représente un tableau d'octets brut (buffer)
    let signature: Data
}
